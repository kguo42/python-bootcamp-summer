cheb = models.Chebyshev1D(degree=3)
cheb_cont = linfitter(cheb, wave[fitWaves], cont[fitWaves])

amplitudeInitial = flux[fitWaves].max()*0.5
o2a = models.Gaussian1D(amplitude=amplitudeInitial, mean=OIIa*(1.+z_init),stddev=3,name='o2a')
o2b = models.Gaussian1D(amplitude=amplitudeInitial, mean=OIIb*(1.+z_init),stddev=3,name='o2b')

o2a.stddev.min = 1.0
o2b.stddev.min = 1.0

combinedFit = o2a + o2b + cheb_cont

#The ratio of O II lines is as expected from atomic physics
#def tie_OIIratio(model):
#    return model.amplitude_1*a3727/a3729
#combinedFit.amplitude_0.tied = tie_OIIratio

#The dispersion of the OII lines should be the same.
def tie_sigma(model):
    return model.stddev_0
combinedFit.stddev_1.tied = tie_sigma

#The redshifts of the lines should all be the same.
def tie_waveA(model):
    return model.mean_0 * OIIb/OIIa
combinedFit.mean_1.tied = tie_waveA

#Don't change redrock z
combinedFit.mean_0.fixed=True
combinedFit.mean_1.fixed=True

fitted_model = fitter(combinedFit,wave[fitWaves], flux[fitWaves], maxiter=500)
